-- 多次点击 Dock 里的微信开发者工具后，会多出一个僵死的，以下可以找到它，但是杀不死……
for i,v in ipairs(hs.application.runningApplications()) do
  local title = v:title()
  if title == 'wechatwebdevtools' then
    local wins = v:allWindows()
    print(i, title, #wins)
    for a,b in ipairs(wins) do
      print(a, b:role(), b:title())
    end
    v:kill()
    v:kill9()
  end
end
